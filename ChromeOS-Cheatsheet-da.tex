\subsection*{ChromwVox-key}
ChromeOS has a screenreader called ChromeVox. ChromeVox uses the ChromeVox-key for working with the screenreader. 
\\ \\
\command{ChromeVox-key (CV)}{\keys{\faSearch} }

\subsection*{Starting and stopping Windows Narrator}
\command{Start and stop \emph{Narrator}}{\keys{$\boxplus$ + \ctrl + \enter}}

\subsection*{Narrator Help}
\command{Start Narrator Command list}{\keys{Na + F1}}
\command{Start Narrator Input-help}{\keys{Na + 1}}
\command{Start Narrator Quickstart}{\keys{Na + something}}

\subsection*{Settings}
\command{Open Settings}{\keys{$\boxplus$ + i}}
\command{Open Accessibility Settings}{\keys{$\boxplus$ + u}}

\subsection*{Adjust Voice-settings}
\command{Faster speech}{\keys{NA + {+}}}
\command{Slower speech}{\keys{NA + {-}}}
\command{Adjust verbosity}{\keys{Na + v}}

\subsection*{Reading commands}
\command{Read line}{\keys{Na + \arrowkeyup}}
\command{Read element from current position}{\keys{Na + \arrowkeydown}}
\command{Read focus}{\keys{Na + \tab}}
\command{Read text-attributes}{\keys{Na + f}}
\command{Stop speech}{\keys{\ctrl}}

\subsection*{Basic navigation}
\command{Move to next element}{\keys{Na + \arrowkeyright}}
\command{Move to previous element}{\keys{Na + \arrowkeyleft}}

\subsection*{Navigating the user-interface}
\command{Goto Notifications}{\keys{$\boxplus$ + b}}
\command{Goto Desktop}{\keys{$\boxplus$ + d}}
\command{Open Activities}{\keys{$\boxplus$ + a}}

\subsection*{Using Scanmode}
\command{Start and stop scanmode}{\keys{Na + \SPACE}}
\command{Jump to next heading}{\keys{h}}
\command{Jump to previous heading}{\keys{\shift + h}}
\command{Jump to next link}{\keys{l}}
\command{Jump to previous link}{\keys{\shift + h}}
\command{Jump to next button}{\keys{b}}
\command{Jump to previous button}{\keys{\shift + b}}
\command{Jump to next editfield}{\keys{e}}
\command{Jump to previous editfield}{\keys{\shift + e}}
\command{Jump to next graphic}{\keys{g}}
\command{Jump to previous graphic}{\keys{\shift + g}}
\command{Describe graphic}{\keys{Na + \ctrl + d}}

\subsection*{Lists}
\command{List of landmarks}{\keys{Na + F5}}
\command{List of headings}{\keys{Na + F6}}
\command{List of links}{\keys{Na + F7}}
