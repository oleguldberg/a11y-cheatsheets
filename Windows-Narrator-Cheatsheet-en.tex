\subsection*{Narrator-keys}
Narrator uses the Narrator-key for working with the screenreader. 
\\ \\
\command{Narrator-key (Na)}{\keys{Capslock} and (or) \keys{Insert}}

\subsection*{Starting and stopping Windows Narrator}
\command{Start and stop \emph{Narrator}}{\keys{\faWindows + \ctrl + \enter}}

\subsection*{Narrator Help}
\command{Start Narrator Command list}{\keys{Na + F1}}
\command{Start Narrator Input-help}{\keys{Na + 1}}
% \command{Start Narrator Quickstart}{\keys{Na + something}}

\subsection*{Settings}
\command{Open Settings}{\keys{\faWindows + i}}
\command{Open Accessibility Settings}{\keys{\faWindows + u}}
\command{Open Narrator Settings}{\keys{\faWindows + \ctrl + n}}


\subsection*{Adjust Voice-settings}
\command{Faster speech}{\keys{Na + {+}}}
\command{Slower speech}{\keys{Na + {-}}}
\command{Adjust verbosity}{\keys{Na + v}}
